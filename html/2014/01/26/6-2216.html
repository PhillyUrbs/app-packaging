Note

There's a mandatory AIM 6.8. update.  I did it with AutoIt.  You can run this without logging in.  After you're done and login to AIM, it asks you to install a second security update!  Sandboxie is also a good way to quickly see file/registry changes.


run("C:\Documents and Settings\Administrator\Install_AIM.exe")


winwait("AIM 6 Installation", "I Agree")

controlclick("AIM 6 Installation", "I Agree", "I Agree")


winwait("AIM 6 Installation", "Installation Options")

controlcommand("AIM 6 Installation", "Installation Options", "Custom Installation", "Check")

controlcommand("AIM 6 Installation", "Installation Options", "Make AIM Search my default search engine", "Uncheck")

controlcommand("AIM 6 Installation", "Installation Options", "Make aol.com my home page", "Uncheck")

controlcommand("AIM 6 Installation", "Installation Options", "Install AIM Toolbar", "Uncheck")

controlclick("AIM 6 Installation", "Installation Options", "Next")


winwait("AIM 6 Installation", "Additional Installation Options")

controlclick("AIM 6 Installation", "Additional Installation Options", "Next")


winwait("AIM 6 Installation", "Installation Complete")

controlclick("AIM 6 Installation", "Installation Complete", "Close")


DirRemove("C:\Documents and Settings\All Users\Start Menu\Programs\AIM", 1)

filedelete("C:\Documents and Settings\All Users\Desktop\AIM 6.lnk")


run("c:\windows\system32\sc.exe stop ""viewpoint manager service""")

run("c:\windows\system32\sc.exe config ""viewpoint manager service"" start= disabled")


; a new update already!

;http://www.aolcdn.com/aolhelp/radioupd.exe

run("c:\documents and settings\administrator\radioupd.exe")

winwait("AOL AmpX Security Update Setup: Installation Options", "Select")

controlclick("AOL AmpX Security Update Setup: Installation Options", "Select", "&Install")

winwait("AOL AmpX Security Update Setup: Completed", "Completed")

controlclick("AOL AmpX Security Update Setup: Completed", "Completed", "&Close")
